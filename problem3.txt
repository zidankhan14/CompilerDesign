#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    string input;
    cout<<"give input: ";
    getline(cin,input);

    if(input.substr(0,2)=="//"){
        cout<<"single line comment"<<endl;
    }
    else if (input.substr(0,2)=="/*" && input.substr(input.length() - 2, 2)=="*/"){
        cout<<"multi line comment"<<endl;
    }
    else{
        cout<<"not a comment"<<endl;
    }

    return 0;
}
