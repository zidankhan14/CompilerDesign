#include<iostream>
using namespace std;

int main() {
   string giveInput;
   cout<<"please type: "<<endl;
   cin>>giveInput;

   cout<<"operators are:"<<endl;

   for(int i=0; i<giveInput.length(); i++){
    if(giveInput[i]=='+' || giveInput[i]=='-' || giveInput[i]=='*' || giveInput[i]=='/' || giveInput[i]=='%' || giveInput[i]=='='){
        cout<<"operator"<<i+1<<":"<<giveInput[i]<<endl;
    }
   }

    return 0;
}
